
# Notes from:
# https://docs.rockylinux.org/guides/import_rocky_to_wsl_howto/

# 1.) use minimal image
# 2.) Boot & install on a new VirtualBox VM. 
# 3.) Install packages:
dnf -y update  # Was not included in Rocky instructions

# From Rocky instructions
# sudo yum install gcc make kernel-devel bzip2 binutils patch libgomp glibc-headers glibc-devel kernel-headers elfutils-libelf-devel tar

# From: https://kifarunix.com/install-virtualbox-guest-additions-on-rocky-linux-8/
dnf install epel-release -y
dnf install dkms kernel-devel kernel-headers gcc make bzip2 perl elfutils-libelf-devel

#              gcc make kernel-devel bzip2 binutils patch libgomp glibc-headers glibc-devel kernel-headers elfutils-libelf-devel tar

# Used superset for packages to install:
dnf -y install gcc make kernel-devel bzip2 binutils patch libgomp glibc-headers glibc-devel kernel-headers elfutils-libelf-devel tar bzip2 dkms perl wget

reboot

# Confirm kernel version installed matches running kernel
rpm -q kernel-devel
uname -r

# If the above don't match, run dnf update kernel-*  (In my case they matched)

# Determine the version of VirtualBox running on W10.  Mine was 6.1.20.
# Download the matching Additions ISO directly from Virtualbox.org:
wget http://download.virtualbox.org/virtualbox/6.1.20/VBoxGuestAdditions_6.1.20.iso -P /tmp

mount /tmp/VBoxGuestAdditions_6.1.20.iso /mnt

visudo   # add admin user to ALL group

# Took snapshot & rebooted, needed to mount again
./VBoxLinuxAdditions.run
reboot

# As standard user, resume Rocky instructions
mkdir wsl_tar
cd wsl_tar
cp -p /etc/yum.conf .
cp -p /etc/yum.repos.d/Rocky-BaseOS.repo .

# Edit wsl_tar to add line:   reposdir=/home/admin/wsl_tar/
